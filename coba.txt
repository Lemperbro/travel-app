    const textarea = document.getElementById('my-textarea');
    
    textarea.addEventListener('input', () => {
      const text = textarea.value;
      const lines = text.split('\n');
      let formattedText = '';
      let inList = false;
      for (let i = 0; i < lines.length; i++) {
        let line = lines[i];
        if (line.trim().startsWith('-')) {
          // Jika baris diawali dengan tanda strip (-), format sebagai item list dengan lingkaran
          line = `<li style="list-style-type: circle;">${line.trim().substr(1)}</li>`;
          // Jika belum dalam list, buat tag <ul>
          if (!inList) {
            line = '<ul>' + line;
            inList = true;
          }
        } else {
          // Jika tidak diawali dengan tanda strip (-), buat tag </ul> jika dalam list
          if (inList) {
            line = '</ul>' + line;
            inList = false;
          }
          // Tambahkan baris tanpa format
          line = `<li>${line.trim()}</li>`;
        }
        formattedText += line;
      }
      // Jika dalam list saat selesai mengetik, buat tag </ul> terakhir
      if (inList) {
        formattedText += '</ul>';
      }
      // Tampilkan teks yang sudah diformat
      textarea.innerHTML = formattedText;
    });
    </script>
